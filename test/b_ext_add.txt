Nx := 128
Ny := 64
setgridsize(Nx, Ny, 1)

c := 5e-9
setcellsize(c, c, c)

Msat = 800e3
Aex  = 13e-12
m    = uniform(1, 0, 0)

mask := newSlice(3, Nx, Ny, 1)

wireX := 0e-9
wireY := 0e-9
wireDiam := 50e-9

for i:=0; i<Nx; i++{
	for j:=0; j<Ny; j++{

		r := index2coord(i, j, 0)
		r = r.sub(vector(wireX, wireY, 0)) // relative position to wire

		B := vector(0, 0, 0)

		if r.len() >= wireDiam{
			current := vector(0, 0, 1)
			B = r.cross(current).mul(mu0 / (2*pi*pow(r.len(), 2)))
		}else{
			relDist := r.len() / wireDiam
			current := vector(0, 0, relDist * relDist)
			b = r.cross(current).mul(mu0 / (2*pi*pow(r.len(), 2)))
		}

		mask.set(0, i, j, 0, b.X())
		mask.set(1, i, j, 0, b.Y())
		mask.set(2, i, j, 0, b.Z())
	}
}


B_ext.add(mask, 0.01)
save(B_ext)

alpha = 3
MaxErr = 1e-4
RunWhile(MaxTorque > 1e-4)

print(average(m))
save(m)

interactive()
